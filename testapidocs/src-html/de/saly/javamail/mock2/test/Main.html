020    public class Main {
021    
022        private static class IdleMessageCountListener implements MessageCountListener {
023    
024            private int addedCount;
025            private int removedCount;
026    
027            protected int getAddedCount() {
028                return addedCount;
029            }
030    
031            protected int getRemovedCount() {
032                return removedCount;
033            }
034    
035            @Override
036            public void messagesAdded(final MessageCountEvent e) {
037                addedCount++;
038    
039            }
040    
041            @Override
042            public void messagesRemoved(final MessageCountEvent e) {
043                removedCount++;
044    
045            }
046    
047        }
048    
049        private static class IdleThread extends Thread {
050            private Exception exception;
051            private final Folder folder;
052            private int idleCount;
053    
054            public IdleThread(final Folder folder) {
055                super();
056                this.folder = folder;
057            }
058    
059            protected Exception getException() {
060                return exception;
061            }
062    
063            protected int getIdleCount() {
064                return idleCount;
065            }
066    
067            @Override
068            public void run() {
069    
070                while (!Thread.interrupted()) {
071                    try {
072                        // System.out.println("enter idle");
073                        ((IMAPFolder) folder).idle();
074                        idleCount++;
075                        // System.out.println("leave idle");
076                        Thread.sleep(500);
077                    } catch (final Exception e) {
078                        exception = e;
079                    }
080                }
081    
082                // System.out.println("leave run()");
083            }
084        }
085    
086        protected static Properties getProperties() {
087    
088            final Properties props = new Properties();
089            props.setProperty("mail.store.protocol", "mock_imaps");
090            return props;
091        }
092    
093        public static void main(final String[] args) throws Exception {
094    
095            final Session session = Session.getInstance(getProperties());
096    
097            final MockMailbox mb = MockMailbox.get("hendrik@unknown.com");
098            final MailboxFolder mf = mb.getInbox();
099    
100            final MimeMessage msg = new MimeMessage((Session) null);
101            msg.setSubject("Test");
102            msg.setFrom("from@sender.com");
103            msg.setText("Some text here ...");
104            msg.setRecipient(RecipientType.TO, new InternetAddress("hendrik@unknown.com"));
105            mf.add(msg); // 11
106            mf.add(msg); // 12
107            mf.add(msg); // 13
108            mb.getRoot().getOrAddSubFolder("test").create().add(msg);
109    
110            final Store store = session.getStore("mock_imap");
111            store.connect("hendrik@unknown.com", null);
112            final Folder defaultFolder = store.getDefaultFolder();
113            final IMAPFolder test = (IMAPFolder) defaultFolder.getFolder("test");
114    
115            final IdleMessageCountListener mcl = new IdleMessageCountListener();
116            test.addMessageCountListener(mcl);
117    
118            test.open(Folder.READ_WRITE);
119    
120            final IdleThread it = new IdleThread(test);
121            it.start();
122    
123            final IdleThread it2 = new IdleThread(test);
124            it2.start();
125    
126            test.addMessages(new Message[] { msg });
127            test.addMessages(new Message[] { msg });
128            test.addMessages(new Message[] { msg });
129    
130            // test.close(true);
131    
132            System.out.println(it.getException());
133            System.out.println(mcl.getAddedCount());
134            System.out.println(mcl.getRemovedCount());
135            System.out.println(test.getMessageCount());
136            System.out.println(it.getIdleCount());
137    
138            // System.exit(1);
139    
140        }
