325        @Test
326        public void testRenameWithSubfolder() throws Exception {
327    
328            // final MockMailbox mb = MockMailbox.get("hendrik@unknown.com");
329    
330            final MimeMessage msg = new MimeMessage((Session) null);
331            msg.setSubject("Test");
332            msg.setFrom("from@sender.com");
333            msg.setText("Some text here ...");
334            msg.setRecipient(RecipientType.TO, new InternetAddress("hendrik@unknown.com"));
335    
336            final Store store = session.getStore("mock_imap");
337            store.connect("hendrik@unknown.com", null);
338            final Folder root = store.getDefaultFolder();
339            final Folder level1 = root.getFolder("LEVEL1");
340            level1.create(Folder.HOLDS_FOLDERS | Folder.HOLDS_MESSAGES);
341            final Folder level2 = level1.getFolder("LEVEL2");
342            level2.create(Folder.HOLDS_FOLDERS | Folder.HOLDS_MESSAGES);
343            level1.appendMessages(new Message[] { msg, msg });
344            level2.appendMessages(new Message[] { msg });
345    
346            Assert.assertTrue(level1.exists());
347            Assert.assertEquals("LEVEL1", level1.getFullName());
348            Assert.assertEquals("LEVEL1", level1.getName());
349            Assert.assertEquals(2, level1.getMessageCount());
350    
351            Assert.assertTrue(level2.exists());
352            Assert.assertEquals("LEVEL1/LEVEL2", level2.getFullName());
353            Assert.assertEquals("LEVEL2", level2.getName());
354            Assert.assertEquals(1, level2.getMessageCount());
355            Assert.assertEquals(2, root.list().length);
356    
357            // getFolder creates a store
358            level1.renameTo(store.getFolder("LEVEL-1R"));
359    
360            // TODO really need a create?
361            Assert.assertTrue(!store.getFolder("LEVEL1").exists());
362    
363            Assert.assertTrue(level1.exists());
364            Assert.assertEquals("LEVEL-1R", level1.getFullName());
365            Assert.assertEquals("LEVEL-1R", level1.getName());
366            Assert.assertEquals(2, level1.getMessageCount());
367    
368            Assert.assertTrue(level2.exists());
369            Assert.assertEquals("LEVEL-1R/LEVEL2", level2.getFullName());
370            Assert.assertEquals("LEVEL2", level2.getName());
371            Assert.assertEquals(1, level2.getMessageCount());
372    
373            Assert.assertEquals(2, root.list().length);
374        }
