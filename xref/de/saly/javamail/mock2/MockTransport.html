26  package de.saly.javamail.mock2;
27  
28  import javax.mail.Address;
29  import javax.mail.Message;
30  import javax.mail.MessagingException;
31  import javax.mail.Session;
32  import javax.mail.Transport;
33  import javax.mail.URLName;
34  import javax.mail.event.ConnectionEvent;
35  import javax.mail.event.TransportEvent;
36  import javax.mail.internet.MimeMessage;
37  
38  public class MockTransport extends Transport {
39  
40      final org.slf4j.Logger logger = org.slf4j.LoggerFactory.getLogger(this.getClass());
41  
42      public MockTransport(final Session session, final URLName urlname) {
43          super(session, urlname);
44  
45      }
46  
47      @Override
48      public void connect(final String host, final int port, final String user, final String password) throws MessagingException {
49  
50          if (isConnected()) {
51              throw new IllegalStateException("already connected");
52          }
53  
54          setConnected(true);
55  
56          notifyConnectionListeners(ConnectionEvent.OPENED);
57      }
58  
59      @Override
60      public void sendMessage(final Message msg, final Address[] addresses) throws MessagingException {
61          for (final Address a : addresses) {
62              final MockMailbox mailbox = MockMailbox.get(a);
63              if (mailbox.getInbox().isSimulateError()) {
64  
65                  notifyTransportListeners(TransportEvent.MESSAGE_NOT_DELIVERED, new Address[0], addresses, new Address[0], msg);
66  
67                  throw new MessagingException("Simulated error sending message to " + a);
68  
69              }
70  
71              mailbox.getInbox().add(new MimeMessage((MimeMessage) msg));
72              notifyTransportListeners(TransportEvent.MESSAGE_DELIVERED, addresses, new Address[0], new Address[0], msg);
73          }
74      }
75  
76  }
