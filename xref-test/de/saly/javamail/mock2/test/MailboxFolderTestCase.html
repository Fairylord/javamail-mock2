60      @Test
61      public void testAddMessages() throws Exception {
62  
63          final MockMailbox mb = MockMailbox.get("hendrik@unknown.com");
64          final MailboxFolder mf = mb.getInbox();
65  
66          final MimeMessage msg = new MimeMessage((Session) null);
67          msg.setSubject("Test");
68          msg.setFrom("from@sender.com");
69          msg.setText("Some text here ...");
70          msg.setRecipient(RecipientType.TO, new InternetAddress("hendrik@unknown.com"));
71          mf.add(msg);
72  
73          Assert.assertEquals(1, mf.getMessageCount());
74          Assert.assertNotNull(mf.getByMsgNum(1));
75          Assert.assertEquals(msg.getSubject(), mf.getByMsgNum(1).getSubject());
76  
77          mf.add(msg);
78          mf.add(msg);
79          Assert.assertEquals(3, mf.getMessageCount());
80          Assert.assertNotNull(mf.getByMsgNum(3));
81          Assert.assertEquals(msg.getSubject(), mf.getByMsgNum(3).getSubject());
82  
83      }

137
