50      @Test
51      public void testAddMessages() throws Exception {
52  
53          final MockMailbox mb = MockMailbox.get("hendrik@unknown.com");
54          final MailboxFolder mf = mb.getInbox();
55  
56          final MimeMessage msg = new MimeMessage((Session) null);
57          msg.setSubject("Test");
58          msg.setFrom("from@sender.com");
59          msg.setText("Some text here ...");
60          msg.setRecipient(RecipientType.TO, new InternetAddress("hendrik@unknown.com"));
61          mf.add(msg); // 11
62          mf.add(msg); // 12
63          mf.add(msg); // 13
64  
65          final Store store = session.getStore("pop3s");
66          store.connect("hendrik@unknown.com", null);
67          final Folder inbox = store.getFolder("INBOX");
68          inbox.open(Folder.READ_ONLY);
69          Assert.assertEquals(3, inbox.getMessageCount());
70          Assert.assertNotNull(inbox.getMessage(1));
71  
72          inbox.close(true);
73  
74          Assert.assertEquals(3, inbox.getMessageCount());
75  
76          inbox.open(Folder.READ_ONLY);
77          inbox.getMessage(1).setFlag(Flag.DELETED, true);
78  
79          inbox.close(true);
80          inbox.open(Folder.READ_ONLY);
81          Assert.assertEquals(2, inbox.getMessageCount());
82          Assert.assertTrue(inbox instanceof POP3Folder);
83          Assert.assertEquals("12", ((POP3Folder) inbox).getUID(inbox.getMessage(1)));
84          inbox.close(true);
85      }

87      @Test
88      public void testDefaultFolder() throws Exception {
89  
90          final MockMailbox mb = MockMailbox.get("hendrik@unknown.com");
91          final MailboxFolder mf = mb.getInbox();
92  
93          final MimeMessage msg = new MimeMessage((Session) null);
94          msg.setSubject("Test");
95          msg.setFrom("from@sender.com");
96          msg.setText("Some text here ...");
97          msg.setRecipient(RecipientType.TO, new InternetAddress("hendrik@unknown.com"));
98          mf.add(msg); // 11
99          mf.add(msg); // 12
100         mf.add(msg); // 13
101         mb.getRoot().getOrAddSubFolder("test").create().add(msg);
102 
103         final Store store = session.getStore("pop3");
104         store.connect("hendrik@unknown.com", null);
105         final Folder defaultFolder = store.getDefaultFolder();
106         final Folder inbox = defaultFolder.getFolder("INBOX");
107 
108         inbox.open(Folder.READ_WRITE);
109 
110         Assert.assertEquals("[INBOX]", Arrays.toString(defaultFolder.list()));
111 
112         Assert.assertEquals(3, inbox.getMessageCount());
113         Assert.assertNotNull(inbox.getMessage(1));
114 
115         inbox.close(true);
116 
117         Assert.assertEquals(3, inbox.getMessageCount());
118 
119         inbox.open(Folder.READ_ONLY);
120         inbox.getMessage(1).setFlag(Flag.DELETED, true);
121 
122         inbox.close(true);
123         inbox.open(Folder.READ_WRITE);
124         Assert.assertEquals(2, inbox.getMessageCount());
125         Assert.assertTrue(inbox instanceof POP3Folder);
126         Assert.assertEquals("12", ((POP3Folder) inbox).getUID(inbox.getMessage(1)));
127         inbox.close(true);
128     }
